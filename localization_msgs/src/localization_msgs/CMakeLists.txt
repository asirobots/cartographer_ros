  "msg/Pose2DRelativeStamped.msg"
  "msg/Pose2DWithCovarianceRelativeStamped.msg"
  "msg/PoseWithCovarianceLeanRelativeStamped.msg"
)

rosidl_generate_interfaces(${PROJECT_NAME}
  ${msg_files}
  DEPENDENCIES builtin_interfaces geometry_msgs std_msgs
